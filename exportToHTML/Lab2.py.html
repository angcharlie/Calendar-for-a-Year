12   def day_of_week(year): 
13       """ 
14       This function takes in an input for year and it returns what day of the week the year starts in Jan 
15       :param year: 
16       :return dayOfWeek: 0 - 6 to correlate to Sunday - Saturday 
17       """ 
18    
19       dayOfWeek = 0   # 0(Sunday), 1(Monday), 2(Tuesday), 3(Wednesday), 4(Thursday), 5(Friday), 6(Saturday) 
20       dayOfWeek = ((year + int((year - 1) / 4) - int((year - 1) / 100) + int((year - 1) / 400)) % 7)  # January 1 in year begins on day 
21       return dayOfWeek

23    
24   def is_leap_year(year): 
25       """ 
26       This function takes in an input for year and it returns whether that year is a leap year or not. 
27       :param year: 
28       :return leapYear: true if leap year, false otherwise 
29       """ 
30    
31       leapYear = True                                                            # boolean variable to see if leap year or not 
32       # leap year if year is divisible by 4 and not by 100 or year is divisible by 400 
33       if (((year % 4) == 0) & ((year % 100) != 0)) | ((year % 400) == 0): 
34           return leapYear 
35       else: 
36           leapYear = False 
37           return leapYear
